SET(PROGS
  driver
  eigenbases
  test_wrcs)

SET(VTK_PROGS
  smp
  fcrit
  ysurf
  invariants04)

SET(TESTS
  invariants01
  invariants02
  invariants03
  invariants05)

FOREACH(var ${PROGS})
    ADD_EXECUTABLE        (${var} "${var}.cpp")
    TARGET_LINK_LIBRARIES (${var} ${LIBS})
    SET_TARGET_PROPERTIES (${var} PROPERTIES COMPILE_FLAGS "${FLAGS}" LINK_FLAGS "${LFLAGS}")
ENDFOREACH(var)

if(A_USE_VTK)
    FOREACH(var ${VTK_PROGS})
        ADD_EXECUTABLE        (${var} "${var}.cpp")
        TARGET_LINK_LIBRARIES (${var} ${LIBS})
        SET_TARGET_PROPERTIES (${var} PROPERTIES COMPILE_FLAGS "${FLAGS}" LINK_FLAGS "${LFLAGS}")
    ENDFOREACH(var)
endif(A_USE_VTK)

FOREACH(var ${TESTS})
    ADD_EXECUTABLE        (${var} "${var}.cpp")
    ADD_TEST              (${var} ${var})
    TARGET_LINK_LIBRARIES (${var} ${LIBS})
    SET_TARGET_PROPERTIES (${var} PROPERTIES COMPILE_FLAGS "${FLAGS}" LINK_FLAGS "${LFLAGS}")
ENDFOREACH(var)
